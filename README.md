Предназначена для автоматизации работа с медицинским оборудованием, его простую эксплуатацию и обслуживание.

### Модель данных

# Пользователь

class User(Base, UserMixin):
    id: int
    name: str           # Имя пользователя
    email: str          # Email (уникальный)
    lastname: str       # Фамилия
    password: str       # Хеш пароля
    role_id: int        # Внешний ключ к роли
    
# Роль пользователя
class Role(Base):
    id: int
    name: str           # Название роли (admin, user)

# Карточка основного средства (Мед оборудования)
class FixedAssed(Base):
    id: int
    name: str           # Наименование оборудования
    purpose: str        # Назначение (вид оборудования)
    serial_number: int  # Серийный номер (уникальный)
    inventory_number: int # Инвентарный номер (уникальный)
    date_entry: datetime # Дата постановки на учет
    status: str         # Статус оборудования
    write_off: bool     # Списано (да/нет)
    model: str          # Модель оборудования
    manufacturer: str   # Производитель
    time_work: datetime # Время наработки
    downtime: datetime  # Время простоя
    last_maintenance: datetime # Дата последнего ТО
